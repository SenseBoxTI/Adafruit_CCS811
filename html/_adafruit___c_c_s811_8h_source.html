195  // measurement and conditions register
196  struct meas_mode {
197  // reserved : 2
198 
199  /* 0: interrupt mode operates normally
200 * 1: Interrupt mode (if enabled) only asserts the nINT signal (driven low) if
201 the new ALG_RESULT_DATA crosses one of the thresholds set in the THRESHOLDS
202 register by more than the hysteresis value (also in the THRESHOLDS register)
203 */
204  uint8_t INT_THRESH : 1;

206  /* 0: int disabled
207 * 1: The nINT signal is asserted (driven low) when a new sample is ready in
